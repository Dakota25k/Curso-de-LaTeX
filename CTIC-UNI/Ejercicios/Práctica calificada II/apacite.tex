\makeatletter
\def\es@sptext#1{%
	{\es@orddot
		\setbox\z@\hbox{8}\dimen@\ht\z@
		\globaldefs\m@ne
		\csname S@\f@size\endcsname
		\globaldefs\z@
		\edef\@tempa{\def\noexpand\@tempc{#1}%
			\lowercase{\def\noexpand\@tempb{#1}}}\@tempa
		\ifx\@tempb\@tempc
		\fontsize\sf@size\z@
		\selectfont
		\advance\dimen@-1.15ex
		\else
		\fontsize\ssf@size\z@
		\selectfont
		\advance\dimen@-1.5ex
		\fi
		\math@fontsfalse\raise\dimen@\hbox{#1}}}
\def\sptext{\es@sptext}
\makeatother
